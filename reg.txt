name_regex = re.compile(r"([A-Za-z]+\s[A-Za-z]+)") 
address_regex = re.compile(r"\d+\s\w+\s\w+") 
email_regex = re.compile(r"([\w.-]+)@([\w.-]+)(.[\w.]+)") 
phone_regex = re.compile(r"\d{3}[-.\s]?\d{3}[-.\s]?\d{4}") 
ssn_regex = re.compile(r"\d{3}[-.\s]?\d{2}[-.\s]?\d{4}") 
cc_regex = re.compile(r"\d{4}[-.\s]?\d{4}[-.\s]?\d{4}[-.\s]?\d{4}") 
bank_regex = re.compile(r"\d{9,}")
